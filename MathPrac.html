        // API Key stored in JS - calls Groq API directly

        const GROQ_API_KEY = ‘DUMMY_API’_KEY;

        

        let currentProblem = null;

        let answered = false;


        function getSelectedTopics() {

        async function generateProblem() {

            const competition = document.getElementById('competition').value;

            const difficulty = parseInt(document.getElementById('difficulty').value);

            const topics = getSelectedTopics();


            const contentDiv = document.getElementById('content');

            contentDiv.innerHTML = '<div class="loading"><div class="spinner"></div><p>Generating problem...</p></div>';

            contentDiv.classList.remove('empty');


            document.getElementById('generateBtn').disabled = true;

            answered = false;


            try {

                const topicsList = topics.length > 0 ? topics.join(", ") : "Algebra";

                const randomSeed = Math.random().toString(36).substring(7);


                const userPrompt = `Generate a ${competition} problem about ${topicsList} at difficulty ${difficulty}/10. ID: ${randomSeed}


Return valid JSON (no markdown, no code blocks):

{

  "problem": "Math problem with $ for LaTeX",

  "answer": "answer in simplest form",

  "explanation": "step-by-step solution"

}`;


                const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {

                    method: 'POST',

                    headers: {

                        'Authorization': `Bearer ${GROQ_API_KEY}`,

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({

                        model: 'llama-3.3-70b-versatile',

                        messages: [

                            { role: 'user', content: userPrompt }

                        ],

                        temperature: 1.0,

                        top_p: 0.95,

                        max_tokens: 800

                    })

                });


                if (!response.ok) {

                    const errorData = await response.json();

                    throw new Error(errorData.error?.message || 'Failed to generate problem');

                }


                const data = await response.json();

                let content = data.choices[0]?.message?.content;


                if (!content) {

                    throw new Error('No response from AI');

                }


                // Clean markdown

                content = content.replace(/```json\s*/g, '').replace(/```\s*/g, '').trim();


                // Fix unescaped backslashes - replace single \ before non-JSON escape chars with \\

                content = content.replace(/\\(?!["\\/bfnrtu])/g, '\\\\');


                // Parse JSON

                try {

                    currentProblem = JSON.parse(content);

                    if (!currentProblem.problem || !currentProblem.answer || !currentProblem.explanation) {

                        throw new Error('Missing fields');

                    }

                } catch (parseError) {

                    // Extract JSON and retry

                    const match = content.match(/\{[\s\S]*\}/);

                    if (!match) throw new Error('Invalid response');

                    const fixed = match[0].replace(/\\(?!["\\/bfnrtu])/g, '\\\\');

                    currentProblem = JSON.parse(fixed);

                }


                displayProblem();

            } catch (error) {

                console.error('API Error:', error);

                contentDiv.innerHTML = `<div class="error">Error: ${error.message}</div>`;

            } finally {

                document.getElementById('generateBtn').disabled = false;

            }

        }


        function displayProblem() {

            const contentDiv = document.getElementById('content');

            const competition = document.getElementById('competition').value;


            contentDiv.innerHTML = `

                <div class="problem-container">

                    <div class="problem-header">

                        <span class="competition-badge">${competition}</span>

                    </div>

                    <div class="problem-text" id="problemText">${currentProblem.problem}</div>

                    <div class="answer-input-section">

                        <label for="answerInput">Your Answer:</label>

                        <input type="text" id="answerInput" placeholder="Enter your answer" onkeypress="if(event.key==='Enter') submitAnswer()">

                        <div class="action-buttons">

                            <button class="btn btn-submit" onclick="submitAnswer()">Submit</button>

                            <button class="btn btn-skip" onclick="generateProblem()">Skip</button>

                        </div>

                    </div>

                </div>

            `;


            // Render MathJax after DOM update

            setTimeout(() => {

                if (window.MathJax && window.MathJax.typesetPromise) {

                    MathJax.typesetPromise([contentDiv]).catch(x => console.log(x));

                }

                document.getElementById('answerInput').focus();

            }, 10);

        }


        function normalizeAnswer(answer) {

            return answer.toLowerCase().trim()

                .replace(/\s+/g, '')

                .replace(/^[a-e]\)?\s*/i, '')

                .replace(/\\frac/g, '/')

                .replace(/\\sqrt/g, 'sqrt')

                .replace(/[\{\}]/g, '');

        }


        function submitAnswer() {

            if (answered) return;


            const userAnswer = document.getElementById('answerInput').value.trim();

            if (!userAnswer) {

                alert('Please enter an answer');

                return;

            }


            answered = true;

            const isCorrect = normalizeAnswer(userAnswer) === normalizeAnswer(currentProblem.answer);


            const contentDiv = document.getElementById('content');

            const feedbackClass = isCorrect ? 'correct' : 'incorrect';

            const feedbackText = isCorrect ? '✅ Correct!' : `❌ Incorrect. The correct answer is: ${currentProblem.answer}`;


            const explanationHtml = `<div class="explanation"><strong>Solution:</strong><div style="margin-top: 10px;">${currentProblem.explanation}</div></div>`;


            contentDiv.innerHTML = `

                <div class="problem-container">
